Gradle

```gradle
    maven {
        url "https://repo.fantasyrealms.net/releases"
    }
```

Adding the dependency:

```gradle
    dependencies {
        implementation "cc.happyareabean:MojangAPI:3.0.0"
    }
```

```xml
</repository>
 ```

```xml
    <groupId>cc.happyareabean</groupId>
    <artifactId>MojangAPI</artifactId>

</dependency>
```
